\begin{slide}
	\Large{Problem}
	\begin{itemize}
    \item Input: chain of $n$ tasks $P = p_1, p_2, ..., p_n$
    \item Output: $m$ disjoint intervals  $S = I_1, I_2, ..., I_m$ which
    			partitions $[[1..m]]$
    \item Metric: minimize the cost of the longest interval:
    			$\min(\max_{i\in\{1..m\}}(\sum_{j \in{I_i}}^m p_j))$
	\end{itemize}
\end{slide}
